no shutdown
exit

no shutdown
exit
int fa 0/0 
ip address 192.168.2.1 255.255.255.0
no shutdown
exit
int fa 0/1 
ip address 192.168.3.1 255.255.255.0
no shutdown
exit

ip classless
router rip
network 192.168.0.1
network 192.168.1.1
network 192.168.2.1
